
--// Source
function obfuscate(source, VarName, WaterMark)
    warn("Started obfuscate with VM protection")
    local Variable = VarName or "Taurus_"
    local WM

    if source == nil then
        source = [[print("Hello World!")]]
    end

    local ticks = tick()

    -- Original watermark system
    if typeof(WaterMark) == "string" and WaterMark ~= nil then
        WM = "    "..tostring(WaterMark).." | Protected by jerryroblox + VM"
    else
        WM = "    ".."WaterMark".." | Protected by jerryroblox + VM"
    end

    WM = "--[[".."\n".. tostring(WM) .."\n".."]]--".."\n\n"

    -- Original random function
    local random_ = function(length)
        local letters = {"a","b","c","d","e","f","g","h","i","j","k","l","m","n","o","p","q","r","s","t","u","v","w","x","y","z"}
        local code

        if math.random(1,2) == 1 then
            code = ""..letters[math.random(1,26)]
        else
            code = ""..tostring(letters[math.random(1,26)]):upper()
        end

        for i = 1, tonumber(length) do
            if math.random(1,2) == 1 then
                local get_letters = letters[math.random(1,26)]
                if math.random(1,2) == 1 then
                    code = code..tostring(get_letters):upper()
                else
                    code = code..tostring(get_letters)
                end
            else
                code = code..tostring(math.random(0,9))
            end
        end
        return tostring(code)
    end

    -- Original binary conversion
    local StringToBinary = function(String)
        local BinaryString = {}

        for i, Character in ipairs(String:split'') do
            local Binary = ""
            local Byte = Character:byte()
            while Byte > 0 do
                Binary = tostring(Byte % 2) .. Binary
                Byte = math.modf(Byte / 2)
            end
            table.insert(BinaryString, string.format("%.8d", Binary))
        end

        return table.concat(BinaryString, " ")
    end

    -- Original add_binary function
    local add_binary = function(number, s)
        local highest = 1

        local topic = {
            "Deobfuscate?",
            "Hello World!",
            "IronBrew Fork? Nope.",
            "PSU Fork? Nope.",
            "Touch some grass",
            "New update when?",
            "GhostyDuckyy",
            "Free obfuscator!",
            "VM Protected!",
            "Nice try deobfuscating",
            random_(math.random(50,150)),
        }

        for i,v in pairs(topic) do
            if i > highest then
                highest = i
            end
        end

        for i = 1, tonumber(number) do
            local str = [[local ]]..Variable..tostring(random_(math.random(10,12)))..[[ = ]]..'"'..StringToBinary(tostring(topic[math.random(1, tonumber(highest))]))..'"'.."; "
            s = s..str
        end

        return tostring(s)
    end

    -- NEW: XOR Encryption function
    local xorEncrypt = function(str, key)
        local encrypted = {}
        for i = 1, #str do
            local byte = string.byte(str, i)
            local keyByte = string.byte(key, ((i - 1) % #key) + 1)
            table.insert(encrypted, byte ~ keyByte)
        end
        return encrypted
    end

    -- NEW: Generate VM instructions
    local createVMInstructions = function(src)
        local instructions = {}
        for i = 1, #src do
            table.insert(instructions, {
                op = math.random(1, 4),
                val = string.byte(src, i),
                jmp = math.random(1, 100) > 85 and math.random(1, 5) or 0
            })
        end
        return instructions
    end

    -- NEW: VM Runtime generator
    local generateVM = function(varPrefix)
        return [[
local ]]..varPrefix..[[VM = function(inst, key)
    local stack, ptr, reg = {}, 1, {}
    
    local function d(data, k)
        local r = {}
        for i = 1, #data do
            local b = data[i]
            local kb = string.byte(k, ((i - 1) % #k) + 1)
            table.insert(r, string.char(b ~ kb))
        end
        return table.concat(r)
    end
    
    local function chk()
        if not game or not game.GetService then return false end
        local s = pcall(function() return game:GetService("Players") end)
        return s
    end
    
    if not chk() then return end
    
    while ptr <= #inst do
        local i = inst[ptr]
        local op, val, jmp = i.op, i.val, i.jmp
        
        if (val * 2) % 2 == 0 then
            if op == 1 then stack[#stack+1] = val
            elseif op == 2 then reg[#reg+1] = val
            elseif op == 3 then stack[#stack+1] = val
            elseif op == 4 and #stack > 0 then table.remove(stack) end
        end
        
        if jmp > 0 and math.random(1,100) > 95 then
            ptr = math.max(1, ptr - jmp)
        else
            ptr = ptr + 1
        end
        
        if ptr > #inst * 2 then break end
    end
    
    local dec = {}
    for i = 1, #inst do
        table.insert(dec, inst[i].val)
    end
    
    local src = d(dec, key)
    return loadstring(src)()
end
]]
    end

    -- NEW: Anti-debug
    local antiDebug = [[
local ]]..Variable..[[chk = function()
    local c = 0
    if game then c = c + 1 end
    if game.GetService then c = c + 1 end
    if pcall(function() game:GetService("CoreGui") end) then c = c + 1 end
    return c >= 2
end
if not ]]..Variable..[[chk() then return end
]]

    local Random_Variable = {
        TableByte = random_(math.random(15,20)),
        Table_concat = random_(math.random(15,20)),
        Loadstring = random_(math.random(15,20)),
        VMKey = random_(math.random(15,20)),
        VMInst = random_(math.random(15,20)),
    }

    -- Original troll function
    local troll_func = [[function() ]]..add_binary(math.random(30,50), "")..[[ end]]
    local troll_var = [[local ]]..Variable..tostring(random_(math.random(15,20)))..[[ = ]]..troll_func

    -- NEW: Generate encryption key
    local encKey = random_(32)
    
    -- NEW: Create VM instructions
    local vmInstructions = createVMInstructions(source)
    
    -- NEW: Encrypt source
    local encrypted = xorEncrypt(source, encKey)

    -- Original source byte encoding (as backup layer)
    local SourceByte = ""
    for i = 1,string.len(source) do SourceByte = SourceByte..'"\\'..string.byte(source, i)..'", ' end
    local TableByte = [[local ]]..Variable..tostring(Random_Variable.TableByte)..[[ = {]]..SourceByte..[[}]]
    
    -- Original loadstring
    local Loadstring = [[local ]]..Variable..tostring(Random_Variable.Loadstring)..[[ = loadstring(table.concat({"\114", "\101", "\116", "\117", "\114", "\110", "\32", "\102", "\117", "\110", "\99", "\116", "\105", "\111", "\110", "\40", "\98", "\121", "\116", "\101", "\41", "\10", "\32", "\32", "\32", "\32", "\105", "\102", "\32", "\116", "\121", "\112", "\101", "\111", "\102", "\40", "\98", "\121", "\116", "\101", "\41", "\32", "\61", "\61", "\32", "\34", "\116", "\97", "\98", "\108", "\101", "\34", "\32", "\116", "\104", "\101", "\110", "\10", "\32", "\32", "\32", "\32", "\32", "\32", "\32", "\32", "\108", "\111", "\97", "\100", "\115", "\116", "\114", "\105", "\110", "\103", "\40", "\116", "\97", "\98", "\108", "\101", "\46", "\99", "\111", "\110", "\99", "\97", "\116", "\40", "\98", "\121", "\116", "\101", "\41", "\41", "\40", "\41", "\10", "\32", "\32", "\32", "\32", "\101", "\108", "\115", "\101", "\10", "\32", "\32", "\32", "\32", "\32", "\32", "\32", "\32", "\98", "\121", "\116", "\101", "\32", "\61", "\32", "\123", "\98", "\121", "\116", "\101", "\125", "\10", "\32", "\32", "\32", "\32", "\32", "\32", "\32", "\32", "\108", "\111", "\97", "\100", "\115", "\116", "\114", "\105", "\110", "\103", "\40", "\116", "\97", "\98", "\108", "\101", "\46", "\99", "\111", "\110", "\99", "\97", "\116", "\40", "\98", "\121", "\116", "\101", "\41", "\41", "\40", "\41", "\10", "\32", "\32", "\32", "\32", "\101", "\110", "\100", "\10", "\101", "\110", "\100", "\10",}))()]]
    
    local func = {
        [1] = Variable..tostring(Random_Variable.Loadstring),
        [2] = Variable..tostring(Random_Variable.TableByte),
    }

    -- Original fake code function
    local fake_code = function(number,r)
        local t = {}
        for i = 1, tonumber(number) do
            local create_Var = Variable..tostring(random_(math.random(15,20)))
            local random

            if r ~= nil then
                random = "return "..tostring(random_(tonumber(r)))
            else
                random = "return "..tostring(random_(math.clamp(1000, string.len(source) / 2, string.len(source))))
            end

            local byte = ""
            for x = 1,string.len(random) do byte = byte..'"\\'..string.byte(random, x)..'", ' end
            local fake = [[local ]]..create_Var..[[ = {]]..byte..[[}; ]]..[[local ]]..create_Var.." = "..func[1]..[[(]]..create_Var..[[); ]]
            table.insert(t,fake)
        end

        return unpack(t)
    end

    -- NEW: Encode VM instructions
    local instStr = "{"
    for i, inst in ipairs(vmInstructions) do
        instStr = instStr .. string.format("{op=%d,val=%d,jmp=%d},", inst.op, inst.val, inst.jmp)
    end
    instStr = instStr .. "}"

    -- NEW: Encode encryption key
    local keyBytes = {}
    for i = 1, #encKey do
        table.insert(keyBytes, string.byte(encKey, i))
    end
    local keyStr = "{" .. table.concat(keyBytes, ",") .. "}"

    -- NEW: VM Runtime
    local vmRuntime = generateVM(Variable)

    -- NEW: Key decoder
    local keyDecoder = [[
local ]]..Variable..Random_Variable.VMKey..[[ = (function()
    local b = ]]..keyStr..[[
    local r = ""
    for i = 1, #b do r = r .. string.char(b[i]) end
    return r
end)()
]]

    -- NEW: Instruction storage
    local instStorage = [[local ]]..Variable..Random_Variable.VMInst..[[ = ]]..instStr

    -- Build final obfuscated code with BOTH layers
    local obfuscated = WM
    obfuscated = obfuscated .. troll_var .. "; "
    obfuscated = obfuscated .. antiDebug
    obfuscated = obfuscated .. add_binary(math.random(20,30), "")
    obfuscated = obfuscated .. vmRuntime
    obfuscated = obfuscated .. Loadstring .. "; "
    obfuscated = obfuscated .. fake_code(math.random(3,5), math.random(400,600))
    obfuscated = obfuscated .. keyDecoder
    obfuscated = obfuscated .. instStorage .. "; "
    obfuscated = obfuscated .. add_binary(math.random(15,25), "")
    obfuscated = obfuscated .. TableByte .. "; "
    obfuscated = obfuscated .. [[local ]]..Variable..tostring(random_(math.random(15,20))).." = "..func[1].."("..func[2]..")" .. "; "
    obfuscated = obfuscated .. fake_code(math.random(3,5), math.random(string.len(source) / 2, string.len(source) * 2))
    obfuscated = obfuscated .. Variable..[[VM(]]..Variable..Random_Variable.VMInst..[[, ]]..Variable..Random_Variable.VMKey..[[)]]
    
    setclipboard(obfuscated)
    warn("Done obfuscate with VM in "..tostring(tick() - ticks).." second")
    return
end

--// Module
return function(source, CustomVarName, WaterMark)
    task.spawn(function()
        obfuscate(source, CustomVarName, WaterMark)
    end)
end
